# don't use the builtin max() function. Afterward, try using max()

def get_max(numbers):
    current_biggest = numbers[0]
    for num in numbers:
        if num > current_biggest:
            current_biggest = num
    return current_biggest

def aggregate_counts(order_list):
    orders_by_count = {}
    for order in order_list:
        if order in orders_by_count:
           orders_by_count[order] += 1
        else:
           orders_by_count[order] = 1
    return orders_by_count

    except ValueError:
        pass

        # let's do a sanity check and exclude everything before 1900
        if sanitized_year > 1900:
            return sanitized_year
